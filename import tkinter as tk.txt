import tkinter as tk

def ChangeLabel():
	label.config(text = entry.get())
	entry.delete(0, len(entry.get()))

window = tk.Tk()
label = tk.Label(window, text = "Hello!")
button = tk.Button(window, text = "Click Here!", command = ChangeLabel)
entry = tk.Entry(window)

label.pack()
button.pack()
entry.pack()

window.mainloop()
